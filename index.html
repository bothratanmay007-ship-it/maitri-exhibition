/* Transport shim: keeps the same google.script.run interface, but sends

   requests to the Apps Script JSON API instead (works from any host). */

function makeRunner() {

  let onOk = function(){}, onFail = function(){};

  const target = {

    withSuccessHandler(f) { onOk = f; return proxy; },

    withFailureHandler(f) { onFail = f; return proxy; }

  };

  const proxy = new Proxy(target, {

    get(t, prop) {

      if (prop in t) return t[prop];

      return function(...args) {

        fetch(API_URL, { method: 'POST', body: JSON.stringify({ fn: prop, args: args }) })

          .then(r => r.json())

          .then(j => { if (j.ok) onOk(j.data); else onFail(new Error(j.error || 'Server error')); })

          .catch(e => onFail(e));

      };

    }

  });

  return proxy;

}

function toggleScanner() {

  const btn = document.getElementById('scan-toggle');

  if (scanning) {

    scanner.stop().then(() => scanner.clear()).catch(() => {});

    scanning = false; btn.textContent = '📷 Start Camera Scan';

    if (wakeLock) { wakeLock.release().catch(()=>{}); wakeLock = null; }

    return;

  }

  scanner = new Html5Qrcode('reader');

  scanner.start(

    { facingMode: 'environment' },

    { fps: 12, qrbox: { width: 250, height: 140 } },

    code => {

      const now = Date.now();

      if (code === lastScan.code && now - lastScan.t < 1500) return; // same-code cooldown only

      lastScan = { code, t: now };

      addToCart(code, 1);

    }

  ).then(() => {

    scanning = true; btn.textContent = '⏹ Stop Camera';

    if ('wakeLock' in navigator) navigator.wakeLock.request('screen').then(w => wakeLock = w).catch(()=>{});

  }).catch(e => toast('Camera error: open directly in Chrome/Safari (not inside another app).'));

}

function addManual() {

  const code = val('s-barcode');

  const qty = Math.max(1, parseInt(document.getElementById('s-qty').value) || 1);

  if (!code) return;

  addToCart(code, qty);

  document.getElementById('s-barcode').value = '';

  document.getElementById('s-qty').value = '1';

}

function addToCart(code, qty) {

  beep();

  const existing = cart.find(i => i.barcode === code);

  if (existing) {

    existing.qty += qty;

    toast((existing.design || code) + ' · ' + existing.qty + ' pcs');

  } else {

    const p = productMap[code];

    cart.push({ barcode: code, design: p ? p.design : '', name: p ? p.name : 'Not in product master',

                qty: qty, sales: document.getElementById('s-sales').value || '', category: p ? p.category : '' });

    toast('✓ ' + (p && p.design ? p.design : code));

  }

  renderCart(); markDirty();

}

function changeQty(i, d) { cart[i].qty = Math.max(1, cart[i].qty + d); renderCart(); markDirty(); }

function setQty(i, v) {

  const n = Math.max(1, parseInt(v) || 1);

  cart[i].qty = n; renderCart(); markDirty();

}

function removeItem(i) { cart.splice(i, 1); renderCart(); markDirty(); }


function renderCart() {

  document.querySelector('#cart tbody').innerHTML = cart.map((it, i) => `

    <tr>

      <td>

        <div class="item-code">${it.design || it.barcode}</div>

        <div class="item-name">${it.design ? it.barcode + (it.name && it.name !== 'Not in product master' ? ' · ' + it.name : '') : it.name}</div>

        ${it.sales ? '<div class="item-sales">' + it.sales + '</div>' : ''}

      </td>

      <td style="white-space:nowrap; text-align:right">

        <button class="qty-btn" onclick="changeQty(${i},-1)">−</button>

        <input class="qty-in" inputmode="numeric" value="${it.qty}" onchange="setQty(${i},this.value)">

        <button class="qty-btn" onclick="changeQty(${i},1)">+</button>

        <button class="del" onclick="removeItem(${i})">✕</button>

      </td>

    </tr>`).join('');

  const total = cart.reduce((s, i) => s + i.qty, 0);

  const label = cart.length + ' designs · ' + total + ' pcs';

  document.getElementById('cart-count').textContent = label;

  document.getElementById('cart-empty').style.display = cart.length ? 'none' : 'block';

  document.getElementById('s-savep').disabled = !cart.length || !selectedParty;

  document.getElementById('s-final').disabled = !cart.length || !selectedParty;

  const strip = document.getElementById('total-strip');

  strip.classList.toggle('show', !!selectedParty && cart.length > 0);

  document.getElementById('strip-party').textContent = selectedParty ? selectedParty.name : '';

  document.getElementById('strip-count').textContent = label;

}

function markDirty() { saveDirty = true; }

function payload() {

function saveProgress(manual) {

  if (!selectedParty || !cart.length) return;

  if (!document.getElementById('s-sales').value || document.getElementById('s-sales').value === '__add__')

    return toast('Select salesperson first');

  if (saveInFlight) { saveDirty = true; return; }

  saveInFlight = true; saveDirty = false;

  syncChip('busy', '● saving…');

  const btn = document.getElementById('s-savep');

  if (manual) { btn.disabled = true; btn.textContent = 'Saving…'; }

  attemptSave(payload(), 0, manual);

}

function attemptSave(pl, tries, manual) {

  google.script.run

    .withSuccessHandler(res => {

      currentOrderId = res.orderId;

      saveInFlight = false;

      syncChip('ok', '✓ synced');

      const btn = document.getElementById('s-savep');

      btn.textContent = 'Save Progress (keep open)'; btn.disabled = !cart.length;

      if (manual) toast('✓ Saved as ' + res.orderId + ' (still open for next department)');

      if (saveDirty) saveProgress(false); // newer changes arrived meanwhile

    })

    .withFailureHandler(() => {

      if (tries < 3) { setTimeout(() => attemptSave(pl, tries + 1, manual), 2000 * (tries + 1)); }

      else {

        saveInFlight = false; saveDirty = true;

        syncChip('err', '⚠ retry pending');

        const btn = document.getElementById('s-savep');

        btn.textContent = 'Save Progress (keep open)'; btn.disabled = false;

        toast('Network issue — your items are safe on this phone. Tap Save again.');

      }

    }).saveOrderProgress(pl);

}

function saveMapping() {

  if (mapSaving) return;

  const code = val('m-code');

  const design = val('m-design');

  if (!code) return toast('Scan or type the QR code first');

  if (!design) return toast('Enter the design number');

  const overwrite = !!(productMap[code] && productMap[code].design);

  mapSaving = true;

  const btn = document.getElementById('m-save');

  btn.disabled = true; btn.textContent = 'Saving…';

  google.script.run

    .withSuccessHandler(res => {

      mapSaving = false; btn.disabled = false; btn.textContent = 'Save Mapping';

      // update local product map + device cache instantly

      productMap[code] = Object.assign(productMap[code] || {}, { design: design, name: productMap[code] ? productMap[code].name : '', category: productMap[code] ? productMap[code].category : '', img: productMap[code] ? productMap[code].img : '' });

      try {

        const c = JSON.parse(localStorage.getItem('ekum_products') || '[]');

        const i = c.findIndex(r => r[0] === code);

        if (i >= 0) c[i][1] = design; else c.push([code, design, '', '', '']);

        localStorage.setItem('ekum_products', JSON.stringify(c));

      } catch(e) {}

      mapCount++;

      document.getElementById('map-count').textContent = mapCount + ' mapped this session';

      const log = document.getElementById('map-log');

      log.className = '';

      log.innerHTML = '<div style="font-size:13px;padding:5px 0;border-bottom:1px solid var(--border)"><b>' + code + '</b> → ' + design +

        (res.updated ? ' <span class="pill open" style="margin-left:6px">updated</span>' : '') + '</div>' + (log.innerHTML.startsWith('<div') ? log.innerHTML : '');

      document.getElementById('m-code').value = '';

      document.getElementById('m-design').value = '';

      document.getElementById('map-note').style.display = 'none';

      toast('✓ ' + code + ' → ' + design);

    })

    .withFailureHandler(e => {

      mapSaving = false; btn.disabled = false; btn.textContent = 'Save Mapping';

      toast('Error: ' + e.message + ' — not saved, try again.');

    }).addProductMapping(code, design, overwrite);

}

/* ============ BOOT / SYNC / AUTO-REFRESH ============ */

function loadProducts() {

  // serve cached copy instantly, refresh in background

  try {

    const c = localStorage.getItem('ekum_products');

    if (c) buildProductMap(JSON.parse(c));

  } catch(e) {}

  google.script.run.withSuccessHandler(rows => {

    buildProductMap(rows);

    try { localStorage.setItem('ekum_products', JSON.stringify(rows)); } catch(e) {}

    toast('Product master loaded: ' + rows.length + ' items');

  }).getProducts();

}

function buildProductMap(rows) {

  productMap = {};

  rows.forEach(r => productMap[r[0]] = { design: r[1], name: r[2], category: r[3], img: r[4] });

}

function fullSync(showToast) {

  syncChip('busy', '● syncing…');

  google.script.run.withSuccessHandler(d => {

    parties = d.parties; orders = d.orders; masters = d.masters;

    sortMasters(); renderMasters(); renderParties(); renderOrders();

    syncChip('ok', '✓ synced');

    if (showToast) toast('Synced');

  }).withFailureHandler(() => syncChip('err', '⚠ offline')).getBootData();

  loadProducts();

}

function lightSync() {

  google.script.run.withSuccessHandler(d => {

    parties = d.parties; masters = d.masters;

    sortMasters(); renderMasters(); renderParties();

    syncChip('ok', '✓ synced');

  }).withFailureHandler(() => syncChip('err', '⚠ offline')).getLightSync();

}

function sortMasters() {

  masters.agents.sort((a,b) => a.localeCompare(b));

  masters.salespersons.sort((a,b) => a.localeCompare(b));

}
